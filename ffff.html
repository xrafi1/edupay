Setup Instructions

Google Apps Script Setup

Create a new Google Apps Script project and replace the default code with the following:

// Replace with your Spreadsheet ID
const SPREADSHEET_ID = "https://script.google.com/macros/s/AKfycbwr-y8mqw6rWH3FEbZHhsJLjjzAHkRcEYDC4i66g60tyc7P-qLkQuc-IK6joJNHakIt/exec";

function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const phoneNumber = data.phoneNumber;
    const selectedSheetName = data.selectedSheetName;
    
    const result = submitDT(phoneNumber, selectedSheetName);
    
    return ContentService
      .createTextOutput(JSON.stringify(result))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (error) {
    return ContentService
      .createTextOutput(JSON.stringify({error: "Invalid request format"}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  const htmlOutput = HtmlService.createHtmlOutputFromFile('index');
  htmlOutput.setTitle("School Result Portal");
  htmlOutput.setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL);
  return htmlOutput;
}

function submitDT(phoneNumber, selectedSheetName) {
  try {
    var ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    var sheet = ss.getSheetByName(selectedSheetName);
    
    if (!sheet) {
      return {error: "Selected class data not found."};
    }
    
    var dataRange = sheet.getRange(2, 1, sheet.getLastRow() - 1, sheet.getLastColumn());
    var dataValues = dataRange.getValues();
    var studentData = null;

    for (var i = 0; i < dataValues.length; i++) {
      var storedPhoneNumber = dataValues[i][1].toString().trim();

      if (storedPhoneNumber === phoneNumber.toString().trim()) {
        // Get max marks from row 11 (assuming they're in columns G to L)
        var maxMarks = sheet.getRange(11, 7, 1, 6).getValues()[0];
        
        studentData = {
          imageUrl: dataValues[i][0] || "https://via.placeholder.com/100",
          roll: dataValues[i][1],
          registration: dataValues[i][2],
          studentName: dataValues[i][3],
          fatherName: dataValues[i][4],
          institution: dataValues[i][5],
          subjects: [
            {name: "Assamese", marks: dataValues[i][6], maxMarks: maxMarks[0]},
            {name: "English", marks: dataValues[i][7], maxMarks: maxMarks[1]},
            {name: "G.Science", marks: dataValues[i][8], maxMarks: maxMarks[2]},
            {name: "G.Mathematics", marks: dataValues[i][9], maxMarks: maxMarks[3]},
            {name: "S.Science", marks: dataValues[i][10], maxMarks: maxMarks[4]},
            {name: "Elective", marks: dataValues[i][11], maxMarks: maxMarks[5]}
          ],
          grade: dataValues[i][15],
          remarks: dataValues[i][16]
        };

        // Calculate totals
        studentData.totalMaxMarks = maxMarks.reduce((a, b) => a + b, 0);
        studentData.totalObtainedMarks = studentData.subjects.reduce((sum, subject) => sum + subject.marks, 0);
        studentData.percentage = ((studentData.totalObtainedMarks / studentData.totalMaxMarks) * 100).toFixed(2);
        
        break;
      }
    }

    if (!studentData) {
      return {error: "Oops, the Admission Number is not found in our records."};
    }
    
    return {success: true, data: studentData};
  } catch (error) {
    return {error: "An error occurred while fetching the result. Please try again."};
  }
}

Deploy the script as a web app with the following settings:

  • Execute as: Me
  • Who has access: Anyone

Google Sheets Setup

Create a Google Sheet with your student data and share it with the Google Apps Script project.

Replace the SPREADSHEET_ID in the code with your own spreadsheet ID.

Your sheet should have the following columns:

  1. Image URL
  2. Roll Number
  3. Registration Number
  4. Student Name
  5. Father's Name
  6. Institution
  7. Assamese Marks
  8. English Marks
  9. G.Science Marks
  10. G.Mathematics Marks
  11. S.Science Marks
  12. Elective Marks
  13. Grade
  14. Remarks
